/**
* @Classname Command
* @Created by 寂然
* @Description 命令抽象类
*/
public abstract class Command {
    //聚合烤肉师傅
    private BarBecuer barBecuer;
    public Command(BarBecuer barBecuer) {
        this.barBecuer = barBecuer;
    }
    //执行命令
    public abstract void executeCommand();
    public BarBecuer getBarBecuer() {
        return barBecuer;
    }
    public void setBarBecuer(BarBecuer barBecuer) {
        this.barBecuer = barBecuer;
    }
}
/**
* @Classname BakeMuttonCommand
* @Created by 寂然
* @Description 具体的烤羊肉串命令
*/
public class BakeMuttonCommand extends Command{
    public BakeMuttonCommand(BarBecuer barBecuer) {
        super(barBecuer);
    }
    @Override
    public void executeCommand() {
        //具体实现调用烤肉师傅的方法
        this.getBarBecuer().bakeMutton();
    }
}
/**
* @Classname BakeChickenWingCommand
* @Created by 寂然
* @Description 具体的烤鸡翅命令
*/
public class BakeChickenWingCommand extends Command{
    public BakeChickenWingCommand(BarBecuer barBecuer) {
        super(barBecuer);
    }
    @Override
    public void executeCommand() {
        this.getBarBecuer().bakeChickenWing();
    }
}
/**
* @Classname Waiter
* @Created by 寂然
* @Description I walk very slowly, but I never walk backwards
*/
public class Waiter {
    private Command command;
    //记录命令（记录订单）
    public void setOrder(Command command){
        this.command = command;
    }
    //通知后厨执行
    public void inform(){
        command.executeCommand();
    }
}

    # 实现方式三：松耦合设计-改进版

    /**
* @Classname Waiter
* @Created by 寂然
* @Description I walk very slowly, but I never walk backwards
*/
public class Waiter {
    // private Command command;
    //增加存放具体命令的容器 菜单有了
    private List<Command> orders = new ArrayList<>();
    //记录命令（记录订单）
    public void setOrder(Command command){
        orders.add(command);
        //记录客户所点菜品的日志
        System.out.println("add on order" + new Date());
    }
    //取消订单的方法
    public void cancelOrder(Command command){
        orders.remove(command);
        System.out.println("cancel the order" + new Date());
    }
    //通知后厨执行
    public void inform(){
        for (Command order : orders) {
            order.executeCommand();
        }
    }
}
/**
* @Classname Client
* @Created by 寂然
* @Description 客户端
*/
public class Client {
    public static void main(String[] args) {
        //开店前准备
        BarBecuer barBecuer = new BarBecuer();
        BakeMuttonCommand command1 = new BakeMuttonCommand(barBecuer);
        BakeMuttonCommand command2 = new BakeMuttonCommand(barBecuer);
        BakeMuttonCommand command3 = new BakeMuttonCommand(barBecuer);
        BakeChickenWingCommand command4 = new BakeChickenWingCommand(barBecuer);
        BakeChickenWingCommand command5 = new BakeChickenWingCommand(barBecuer);
        //开门营业
        Waiter waiter = new Waiter();
        waiter.setOrder(command1);
        waiter.setOrder(command2);
        waiter.setOrder(command3);
        waiter.setOrder(command4);
        waiter.setOrder(command5);
        //点餐完毕，通知后厨
        waiter.inform();
        //客户觉得羊肉串多了，少点一串
        waiter.cancelOrder(command1);
        waiter.inform();
    }
}

    其实已经使用命令模式的思想，完成了一个简单的业务需求

    # 基本介绍

    # 设计模式概念

    命令模式（Command Pattern）：将一个请求封装为一个对象，使用不同参数去表示不同命令，记录请求日志，还可以对请求排队，同时命令模式也支持可撤销的操作

    • 命令模式使得请求发送者和请求接收者消除彼此之间的耦合，实现解耦
    • 在软件设计中，经常需要向某些对象发送请求，但是并不知道请求的接受者是谁，只需要在程序运行的过程中动态的指定具体的请求接收者，这样的情况可以使用命令模式去设计

    # 原理类图

    # 角色分析

    • Invoker ：是调用者角色，即命令的发布者，对应顾客
    • Command ：抽象的命令角色，定义需要执行的所有命令，可以是接口或者抽象类
    • ConcreteCommandA /B：具体的命令角色，继承或者实现Command，将一个接收者对象和一个命令绑定调用接受者相关的操作，对应我们烤羊肉串/鸡翅命令
    • Receiver ：接收者角色，定义如何实施和执行一个命令相关的操作，对应服务员

    # 案例演示 - 智能家电问题

    老张购买了一套智能家电，有照明灯、电视、冰箱、洗衣机，只要在手机上安装 app 就可以控制这些家电工作，但是这些智能家电来自不同的厂家，老张不想针对每一种家电都安装一个 App，分别控制，希望只要一个 app 就可以控制全部智能家电，你需要帮助老张解决问题

/**
 * @author 勤学苦练
 * @date 2021/7/25 11:37
 * @Description 命令类
 */
public interface Command {

    /**执行命令*/
    public void excute();

    /**撤销命令*/
    public void undo();

}
/**
 * @author 勤学苦练
 * @date 2021/7/25 11:44
 * @Description 电灯关闭命令
 */
public class LightOffCommand implements Command {

    private LightReceiver lightReceiver;

    public LightOffCommand(LightReceiver lightReceiver) {
        this.lightReceiver = lightReceiver;
    }

    @Override
    public void excute() {
        lightReceiver.off();
    }

    @Override
    public void undo() {
        lightReceiver.on();
    }
}
/**
 * @author 勤学苦练
 * @date 2021/7/25 11:42
 * @Description 电灯的开灯命令
 */
public class LightOnCommand implements Command {

    /**
     * 调用电灯操作类进行操作
     */
    private LightReceiver lightReceiver;

    /**
     * 构造函数获取电灯操作类对象
     * @param lightReceiver
     */
    public LightOnCommand(LightReceiver lightReceiver) {
        this.lightReceiver = lightReceiver;
    }

    @Override
    public void excute() {
        lightReceiver.on();
    }

    @Override
    public void undo() {
        lightReceiver.off();
    }
}

/**
 * @author 勤学苦练
 * @date 2021/7/25 11:45
 * @Description 电灯命令者
 */
public class LightSender {

    Command[] onCommands;
    Command[] offCommands;

    /**
     * 记录上一次命令，用于撤销
     */
    Command undoCommand;

    /**
     * 初始化命令队列
     */
    public LightSender() {
        onCommands = new Command[5];
        offCommands = new Command[5];

        for (int i=0;i < 5; i++) {
            onCommands[i] = new NoCommand();
            offCommands[i] = new NoCommand();
        }
    }

    /**
     * 设置命令方法
     * @param number
     * @param onCommand
     * @param offCommand
     */
    public void setCommand(int number,Command onCommand,Command offCommand) {
        onCommands[number] = onCommand;
        offCommands[number] = offCommand;
    }

    /**
     * 开启命令
     */
    public void onItem(int number) {
        // 执行命令
        onCommands[number].excute();
        // 记录此次命令用于撤销
        undoCommand = onCommands[number];
    }

    /**
     * 关闭命令
     */
    public void offItem(int number) {
        // 执行命令
        offCommands[number].excute();
        // 记录此次命令用于撤销
        undoCommand = offCommands[number];
    }


    public void undoCommand() {
        // 执行撤销操作
        undoCommand.undo();
    }
}
/**
 * @author 勤学苦练
 * @date 2021/7/25 11:54
 * @Description 调用类
 */
public class Client {

    public static void main(String[] args) {
        // 使用命令模式，完成以上只能家具的操作
        LightReceiver lightReceiver = new LightReceiver();
        LightOnCommand lightOnCommand = new LightOnCommand(lightReceiver);
        LightOffCommand lightOffCommand = new LightOffCommand(lightReceiver);
        LightSender lightSender = new LightSender();
        // 给电灯发送者赋予电灯命令对象，它就可以发送命令，并调用电灯接收者执行了。
        lightSender.setCommand(0,lightOnCommand,lightOffCommand);
        System.out.println("用户按下电灯打开按钮。。。");
        lightSender.onItem(0);
        System.out.println("用户按下电灯关闭按钮");
        lightSender.offItem(0);
        System.out.println("用户撤销了一次操作");
        lightSender.undoCommand();

    }
}

/**
 * @author 勤学苦练

public class TvOffCommand implements Command {

    public void excute() {

    @Override

    # 优势

    • 将发起请求的对象与执行请求的对象解耦合
    • 容易设计一个命令队列
    • 容易实现对命令的撤销和重做

    # 局限性

    • 可能导致某些系统有过多的具体命令类，增加系统的复杂度，这点在使用的时候要注意
    • 合理使用，不要为了用而用

    • ThreadPools(线程池)
