    # 解决方案一：模板模式实现

    /**
* @Classname Employee
* @Created by 寂然
* @Description 抽象的员工类
*/
public abstract class Employee {
    private String name;
    private int sex;
    public final static int MALE = 0;//0代表男性
    public final static int FEMALE = 1;//1代表女性
    private int salary;
    public String getName() {
        return name;
    }
    public void setName(String name) {
        this.name = name;
    }
    public int getSex() {
        return sex;
    }
    public void setSex(int sex) {
        this.sex = sex;
    }
    public int getSalary() {
        return salary;
    }
    public void setSalary(int salary) {
        this.salary = salary;
    }
    //打印的方法
    public final void report(){
        //首先获得员工的基本信息
        String info = "姓名：" + this.name + "\t";
        info = info + "性别：" + (this.sex == FEMALE?"女":"男") + "\t";
        info = info + "薪水：" + this.salary + "\t";
        //获得员工的其他信息
        info = info + this.getOtherInfo();
        System.out.println(info);
    }
    //拼装员工其他信息的方法
    public abstract String getOtherInfo();
}
/**
* @Classname CommonEmployee
* @Created by 寂然
* @Description 普通员工类
*/
public class CommonEmployee extends Employee{
    private String job;public String getJob() {
        return job;
    }
    public void setJob(String job) {
        this.job = job;
    }
    @Override
    public String getOtherInfo() {
        return "工作：" + this.job + "\t";
    }
}
/**
* @Classname Manager

public class Manager extends Employee{

    public String getPerformance() {
        return performance;

    public void setPerformance(String performance) {
        this.performance = performance;
    }
    @Override
    public String getOtherInfo() {
        return "业绩："+ this.performance + "\t";
    }
}

    # 解决方案二：访问的思想

    /**
 * @author 勤学苦练
 * @date 2021/7/25 16:36
 * @Description 访问者接口
 */
public interface IVisitor {

    /**
     * 定义普通员工方法
     * @param commonEmployee
     */
    public void visitor(CommonEmployee commonEmployee);

    /**
     * 定义管理人员方法
     * @param manager
     */
    public void visitor(Manager manager);

}

/**
 * @author 勤学苦练
 * @date 2021/7/25 16:21
 * @Description 员工接口
 */
public abstract class Employee {

    /**0代表男性*/
    private final static int MALE = 0;
    /**1代表女性*/
    private final static int FEMALE = 1;

    private String name;
    private int sex;
    private int salary;

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getSex() {
        return sex;
    }

    public void setSex(int sex) {
        this.sex = sex;
    }

    public int getSalary() {
        return salary;
    }

    public void setSalary(int salary) {
        this.salary = salary;
    }

    public abstract void accept(IVisitor iVisitor);
}
/**
 * @author 勤学苦练
 * @date 2021/7/25 16:27
 * @Description 普通员工类
 */
public class CommonEmployee extends Employee {

    private String job;

    public String getJob() {
        return job;
    }

    public void setJob(String job) {
        this.job = job;
    }

    @Override
    public void accept(IVisitor iVisitor) {
        iVisitor.visitor(this);
    }
}

/**
 * @author 勤学苦练
 * @date 2021/7/25 16:25
 * @Description 经理类
 */
public class Manager extends Employee {
    /**业绩*/
    private String performance;

    public String getPerformance() {
        return performance;
    }

    public void setPerformance(String performance) {
        this.performance = performance;
    }

    @Override
    public void accept(IVisitor iVisitor) {
        iVisitor.visitor(this);
    }
}
/**
 * @author 勤学苦练
 * @date 2021/7/25 16:39
 * @Description 访问者操作对象
 */
public class Visitor implements IVisitor {
    @Override
    public void visitor(CommonEmployee commonEmployee) {
        System.out.println(getCommonEmployeeInfo(commonEmployee));
    }

    @Override
    public void visitor(Manager manager) {
        System.out.println(getMangerInfo(manager));
    }

    /**
     * 打印汇总员工信息
     * @param employee
     * @return
     */
    public String getBaseInfo(Employee employee) {
        // 获取员工基本信息
        String info = "姓名：" + employee.getName() + "\t";
        info += "性别：" + (employee.getSex() == 0 ? "男" : "女") + "\t";
        info += "薪水：" + employee.getSalary() + "\t";

        return info;
    }

    /**
     * 打印管理人员信息的方法
     * @param manager
     * @return
     */
    public String getMangerInfo(Manager manager) {
        String info = "业绩：" + manager.getPerformance() + "\t";
        return getBaseInfo(manager) + info;
    }

    /**
     * 打印普通员工信息的方法
     * @param commonEmployee
     * @return
     */
    public String getCommonEmployeeInfo(CommonEmployee commonEmployee) {
        String info = "工作：" + commonEmployee.getJob() + "\t";
        return getBaseInfo(commonEmployee) + info;
    }

}

/**
 * @author 勤学苦练

 */
public class Client {

    public static void main(String[] args) {

        List<Employee> employeeList = mockEmployee();

    一起分析下这个程序的实现：

    • 通过循环遍历所有元素
    • 每一个员工对象都定义了一个访问者
    • 员工对象把自己作为一个参数调用访问者的visit方法
    • 访问者调用自己内部的计算逻辑完成报表的打印
    • 客户端调用方法，完成报表数据的打印

    # 基本介绍

    # 访问者模式的概念

    访问者模式（Visitor pattern）封装了一些作用于某种数据结构的各元素的操作，它可以再不改变数据结构的前提下，定义作用于这些元素的新的操作，主要是将数据结构和数据操作分离

    # 原理类图

    # 角色分析

    • Visitor ：抽象类或者接口，声明访问者可以访问哪些元素
    • ConcreteVisitor :具体的访问者，他会具体实现访问者访问到一个类后该怎么说，具体的业务逻辑
    • Element ：抽象的被访问者，声明接收哪一类访问者的访问
    • ConcreteElement ：具体的被访问者，具体实现业务逻辑
    • ObjectStructure ：元素的产生者，数据结构，一般是一个容纳多个类/接口的容器。

    # 案例演示 - 中国好嗓门

    现在有一档综艺节目 - 中国好嗓门，观众有场内观众和场外观众，每位歌手表演完毕观众都会对该歌手的表演进行投票（晋级/淘汰）将评价进行反馈，根据上述要求，完成测评系统的编码

    /**
 * @author 勤学苦练
 * @date 2021/7/25 17:07
 * @Description 抽象的访问者对象
 */
public abstract class Action {

    /**
     * 得到场内观众的评测结果
     */
    public abstract void getInnerResult(InnerAudience innerAudience);

    /**
     * 得到场外观众的评测结果
     */
    public abstract void getOuterResult(OuterAudience outerAudience);

}

/**
 * @author 勤学苦练
 * @date 2021/7/25 17:06
 * @Description 抽象的观众
 */
public abstract class Audience {

    /**
     * 提供一个方法，供访问者访问
     */
    public abstract void accept(Action action);

}
/**
 * @author 勤学苦练
 * @date 2021/7/25 17:09
 * @Description
 */
public class InnerAudience extends Audience{
    @Override
    public void accept(Action action) {
        action.getInnerResult(this);
    }
}
/**
 * @author 勤学苦练
 * @date 2021/7/25 17:09
 * @Description
 */
public class OuterAudience extends Audience{
    @Override
    public void accept(Action action) {
        action.getOuterResult(this);
    }
}

/**
 * @author 勤学苦练
 * @date 2021/7/25 17:12
 * @Description  数据结构，管理众多观众
 */
public class ObjectStructure {

    /**
     * 维护一个集合
     */
    private List<Audience> audienceList = new ArrayList<>();

    /**
     * 添加一个观众
     * @param audience
     */
    public void add(Audience audience) {
        audienceList.add(audience);
    }

    /**
     * 删除一个观众
     * @param audience
     */
    public void delete(Audience audience) {
        audienceList.remove(audience);
    }

    /**
     * 显示测评结果
     * @param action
     */
    public void display(Action action) {
        for (Audience audience : audienceList) {
            audience.accept(action);
        }
    }

}
/**
 * @author 勤学苦练
 * @date 2021/7/25 17:10
 * @Description 晋级操作
 */
public class Success extends Action{


    @Override
    public void getInnerResult(InnerAudience innerAudience) {
        System.out.println("场内观众给的评价是晋级");
    }

    @Override
    public void getOuterResult(OuterAudience outerAudience) {
        System.out.println("场外观众给的评价是晋级");
    }
}
/**
 * @author 勤学苦练
 * @date 2021/7/25 17:11
 * @Description 淘汰操作
 */
public class Fail extends Action{
    @Override
    public void getInnerResult(InnerAudience innerAudience) {
        System.out.println("场内观众给出的结果是淘汰");
    }

    @Override
    public void getOuterResult(OuterAudience outerAudience) {
        System.out.println("场外观众给出的结果是淘汰");
    }
}

/**
 * @author 勤学苦练
 * @date 2021/7/25 17:14
 * @Description 测试类
 */
public class Client {

    public static void main(String[] args) {

        ObjectStructure objectStructure = new ObjectStructure();

        // 数据结构中添加观众
        objectStructure.add(new InnerAudience());
        objectStructure.add(new OuterAudience());

        // 数据结构中展示观众结果
        objectStructure.display(new Success());
    }

}
